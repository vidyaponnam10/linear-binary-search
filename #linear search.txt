#linear search
arr = list(map(int,input().split()))
target = int(input())
for i in range(len(arr)):
    if arr[i] == target:
        print("element is found:",i)
        break
# binary search
arr = list(map(int,input().split()))
target = int(input())
left  = 0
right = len(arr)-1
while left <= right:
    mid = (left + right)//2
    if arr[mid] == target:
        print("element found:", mid)
        break
    elif arr[mid] < target:
        left = mid + 1
    else:
        right = mid - 1
